SELECT A.ano_eje,A.sec_ejec,A.sec_func,A.funcion,LTRIM(RTRIM(B.NOMBRE)) funcion_nombre,A.programa,LTRIM(RTRIM(P.NOMBRE)) programa_nombre,A.sub_programa, LTRIM(RTRIM(S.NOMBRE)) sub_programa_nombre,A.act_proy,A.meta,LTRIM(RTRIM(F.NOMBRE)) nombre_meta,A.finalidad,A.unidad_medida, LTRIM(RTRIM(U.NOMBRE)) unidad_medida_nombre FROM META A INNER JOIN FUNCION B ON A.ANO_EJE=B.ANO_EJE AND A.FUNCION=B.FUNCION INNER JOIN PROGRAMA_NOMBRE P on A.ANO_EJE=P.ANO_EJE AND A.PROGRAMA=P.PROGRAMA INNER JOIN SUB_PROGRAMA_NOMBRE S on A.ANO_EJE=S.ANO_EJE AND A.SUB_PROGRAMA=S.SUB_PROGRAMA INNER JOIN FINALIDAD F on A.ANO_EJE=F.ANO_EJE AND A.FINALIDAD=F.FINALIDAD INNER JOIN UNIDAD_MEDIDA U ON A.UNIDAD_MEDIDA = U.UNIDAD_MEDIDA WHERE A.ANO_EJE='2020' AND VAL(A.SEC_EJEC)=996
